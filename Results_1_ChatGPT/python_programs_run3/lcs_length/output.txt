def lcs_length(s, t):
    from collections import defaultdict

    dp = defaultdict(int)

    for i in range(len(s)):
        for j in range(len(t)):
            if s[i] == t[j]:
                dp[i, j] = dp[i - 1, j - 1] + 1
            else:
                dp[i, j] = max(dp[i - 1, j], dp[i, j - 1])

    return dp[len(s) - 1, len(t) - 1]
